// Solution: Doubling
{n >= 0}

i := 0;
{0 <= i & i <= n & 1 = exp(2, i)}
x := 1;
{0 <= i & i <= n & x = exp(2, i)}
do [0 <= i & i <= n & x = exp(2, i)]
  i < n ->
    {0 <= i & i < n & x = exp(2, i)}
    x := 2 * x;
    {0 <= (i + 1) & (i + 1) <= n & x = exp(2, (i + 1))}
    i := i + 1
    {0 <= i & i <= n & x = exp(2, i)}
od
{x = exp(2, n)}

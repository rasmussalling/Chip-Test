// Solution: Factorial
{n >= 0}

i := 0;
{0 <= i & i <= n & fac(i) = 1}
f := 1;
{0 <= i & i <= n & f = fac(i)}
do [0 <= i & i <= n & f = fac(i)]
  i < n ->
    {0 <= i & (i + 1) <= n & f * (i + 1) = fac(i + 1)}
    i := i + 1;
    {0 <= i & i <= n & f * i = fac(i)}
    f := f * i
    {0 <= i & i <= n & f = fac(i)}
od
{f = fac(n)}

// Solution: Sum of squares
{n >= 0}

i := 0;
{i <= n & 0 = i * (i + 1) * (2 * i + 1) / 6}
s := 0;
{i <= n & s = i * (i + 1) * (2 * i + 1) / 6}
do [i <= n & s = i * (i + 1) * (2 * i + 1) / 6]
  i < n ->
    {i < n & s = i * (i + 1) * (2 * i + 1) / 6}
    i := i + 1;
    {i <= n & (s + i * i) = i * (i + 1) * (2 * i + 1) / 6}
    s := s + i * i
    {i <= n & s = i * (i + 1) * (2 * i + 1) / 6}
od
{s = n * (n + 1) * (2 * n + 1) / 6}

// Solution: Triangular Numbers
{i != n & t = i * (i + 1) / 2}
do [t = i * (i + 1) / 2] 
  i != n ->
    {t = i * (i + 1) / 2}
    {t = (i + 1) * i / 2}
    {t = (i + 1) * ((i + 1) - 1) / 2}
    i := i + 1;
    {t = i * (i - 1) / 2}
    {t = i * (i + 1) / 2 - i}
    {t + i = i * (i + 1) / 2}
    t := t + i
    {t = i * (i + 1) / 2}
od
{i = n & t = n * (n + 1) / 2}

// Solution: Sum of Squares
{ i != n & s = i * (i + 1) * (2 * i + 1) / 6}
do [s = i * (i + 1) * (2 * i + 1) / 6]
  i != n ->
    {s + (i + 1) * (i + 1) = (i + 1) * ((i + 1) + 1) * (2 * (i + 1) + 1) / 6}
    {s = (i + 1) * (i + 2) * (2 * i + 3) / 6 - exp(i + 1, 2)}
    i := i + 1;
    {s + i * i = i * (i + 1) * (2 * i + 1) / 6}
    s := s + i * i
    { s = i * (i + 1) * (2 * i + 1) / 6 }
od
{ i = n & s = n * (n + 1) * (2 * n + 1) / 6 }

// Solution: Repeated addition
{n >= 0}
{0 <= n & 0 = a * 0}
i := 0;
{i <= n & 0 = a * i}
r := 0;
{i <= n & r = a * i}
do [i <= n & r = a * i]
  i < n ->
    {(i + 1) <= n & (r + a) = a * (i + 1)}
    r := r + a;
    {(i + 1) <= n & r = a * (i + 1)}
    i := i + 1
    {i <= n & r = a * i}
od
{r = a * n}
